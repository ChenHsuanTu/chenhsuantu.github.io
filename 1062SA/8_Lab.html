library(spdep);library(rgdal);library(ggplot2);library(RColorBrewer);library(GISTools)

TW=readOGR(dsn ="Data", layer = "Popn_TWN", encoding="utf8")
for(i in c(4:15)) TW@data[[i]]=as.numeric(as.character(TW@data[[i]]))
old=TW@data$Age_L65/TW@data$Popn
TW@data$old=old
TWTable=TW@data

TW.nb=poly2nb(TW)
TW.nb.w=nb2listw(TW.nb, zero.policy=T)


lm.palette=colorRampPalette(c( "white","orange","red"), space = "rgb")
spplot(TW, zcol="old", col.regions=lm.palette(20), main="()")

LISA map (p-value < 0.05) ( HH, HL, LH, LL)

LISA=localmoran(old, TW.nb.w, zero.policy=T,alternative ="two.sided")
diff=old - mean(old)
z=LISA[,4]
quad=vector(mode="numeric",length=nrow(LISA))
quad[diff>0 & z>0]=1 # H-H
quad[diff<0 & z>0]=2 # L-L
quad[diff>0 & z<0]=3 # H-L
quad[diff<0 & z<0]=4 # L-H

quad[LISA[, 5]>0.05]=5 #localmoran0.05OK
colors=c("red", "blue", "lightpink", "skyblue2", rgb(.95, .95, .95))
plot(TW, border="grey", col=colors[quad], main = "LISA Map\n")
legend("bottomright",legend=c("High-High","Low-Low","High-Low","Low-High","Not Significant"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)

FDR-corrected LISA map (p-value < 0.05)( HH, HL, LH, LL)

LISA.fdr=localmoran(old, TW.nb.w, zero.policy=T,alternative ="two.sided",p.adjust.method = "fdr")
diff=old - mean(old)
z=LISA.fdr[,4]
quad=vector(mode="numeric",length=nrow(LISA))
quad[diff>0 & z>0]=1 # H-H
quad[diff<0 & z>0]=2 # L-L
quad[diff>0 & z<0]=3 # H-L
quad[diff<0 & z<0]=4 # L-H
quad[LISA.fdr[, 5]>0.05]=5 
plot(TW, border="grey", col=colors[quad], main = "LISA Map(FDR)\n")
legend("bottomright",legend=c("High-High","Low-Low","High-Low","Low-High","Not Significant"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)

Standardized Gi * values (p-value < 0.05)( cluster, non-cluster)

TW.nb.in=include.self(TW.nb)
TW.nb.w.in=nb2listw(TW.nb.in)
LG=as.vector(localG(old,TW.nb.w.in))
TW$LG=LG
quad=rep(2,length(LG)) # Non-cluster
quad[LG>1.65]=1 # Cluster

colors=c("red", "lightgray")
plot(TW, border="darkgrey", col=colors[quad], main = "Cluster Map (Gi*)\n")
legend("bottomright",legend=c("Cluster","Non-cluster"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)

Gi* after FDR

p=pnorm(LG,lower.tail =F)
p=p.adjust(p,"fdr")
quad=rep(2,length(LG)) # Non-cluster
quad[p<0.05]=1  # Cluster
plot(TW, border="darkgrey", col=colors[quad], main = "Cluster Map (Gi* after FDR)\n")
legend("bottomright",legend=c("Cluster","Non-cluster"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)
