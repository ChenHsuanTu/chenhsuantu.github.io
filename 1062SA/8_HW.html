rm(list = ls())

library(spdep);library(rgdal);library(ggplot2);library(RColorBrewer);library(GISTools)

TW=readOGR(dsn ="Data", layer = "Popn_TWN", encoding="utf8")
for(i in c(4:15)) TW@data[[i]]=as.numeric(as.character(TW@data[[i]]))
old=TW@data$Age_L65/TW@data$Popn
TW@data$old=old

TW.nb=poly2nb(TW)
TW.nb.w=nb2listw(TW.nb, zero.policy=T)
TW.nb.in=include.self(TW.nb)
TW.nb.w.in=nb2listw(TW.nb.in)

利用蒙地卡羅模擬，計算Gi*的統計顯著性(p-values)，並繪製以下台灣鄉鎮高齡人口比例的主題圖：
一、 p-values of Gi* values

LG=as.vector(localG(old,TW.nb.w.in))
TW$LG=LG
sim.G=matrix(NA, nrow = 9999, ncol = length(LG))
for(i in 1:9999) sim.G[i,]=localG(sample(old),TW.nb.w.in)
p.val=(colSums(sweep(sim.G,2,LG,'>'))+1)/(nrow(sim.G)+1)
TW$p=p.val

col=colorRampPalette(c("red","orange","white"), space = "rgb")
spplot(TW, zcol="p", col.regions=col(20), main="台灣鄉鎮高齡人口比例\np-values of Gi* values（單尾）")

二、統計顯著的高齡人口群聚鄉鎮 (p-value < 0.05)

quad=rep(2,length(LG)) # Non-cluster
quad[p.val<0.05]=1 # Cluster

colors=c("red", "lightgray")
plot(TW, border="darkgrey", col=colors[quad], main = "Cluster Map (Gi* MC)\n台灣鄉鎮高齡人口比例")
legend("bottomright",legend=c("Cluster","Non-cluster"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)

三、FDR修正統計顯著的高齡人口群聚鄉鎮 (p-value < 0.05)

p.fdr=p.adjust(p.val,"fdr")
quad=rep(2,length(LG)) # Non-cluster
quad[p.fdr<0.05]=1 # Cluster

colors=c("red", "lightgray")
plot(TW, border="darkgrey", col=colors[quad], main = "Cluster Map (Gi* MC after FDR)\n台灣鄉鎮高齡人口比例")
legend("bottomright",legend=c("Cluster","Non-cluster"), fill=colors,bty="n",cex=0.7,y.intersp=1,x.intersp=1)
