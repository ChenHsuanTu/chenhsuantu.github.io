問題三、檢視 Moran scatter plot ，繪製鄉鎮地圖與圖例。

z=c()
for (i in 1:length(dep)){
 lag=c()
 if(any(TW.nb[[i]]==0)){z=c(z,0)}else{
   for (j in TW.nb[[i]]) lag=c(lag,dep[j])
   z=c(z,mean(lag))}
}

lvl=rep(1,length(dep))
lvl[dep>mean(dep)&z>mean(z)]=3
lvl[dep<mean(dep)&z<mean(z)]=5
lvl[z==0]=-1

colorset=c("grey","orange","red","blue")
shade=shading(c(0,2,4), cols=colorset)
