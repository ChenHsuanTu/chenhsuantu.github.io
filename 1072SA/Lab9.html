library(splancs);library(GISTools);library(rgdal);library(raster)
setwd("D:/1072SA/Data")

bnd=read.csv("tpe_sqr_bnd.csv");BND=as.points(bnd[,2], bnd[,3])
p1=read.csv("point1.csv");Pts1=as.points(p1[,1],p1[,2])
p2=read.csv("point2.csv");Pts2=as.points(p2[,1],p2[,2])

TW=readOGR(dsn = ".", layer = "Popn_TWN2", encoding="utf8")

proj=TPE@proj4string

splancs KDE

kde1=kernel2d(Pts1,BND, 1500,1000, 1000,quiet=T)
kde2=kernel2d(Pts2,BND, 1500,1000, 1000,quiet=T)
kde_diff=kde1
kde_diff$z=kde1$z-kde2$z
polymap(BND)
image(kde_diff, add=T)
polymap(BND,add=T)

GISTools + raster KDE

PT1=SpatialPoints(Pts1, proj)
PT2=SpatialPoints(Pts2, proj)

KDE1=raster(kde.points(PT1,3000,n=1000, lims=TPE))
KDE2=raster(kde.points(PT2,3000,n=1000, lims=TPE)) 

KDE.Diff=KDE1-KDE2
plot(KDE.Diff, axes=F,box=F,legend=F)
masker=poly.outer(KDE.Diff,TPE)
add.masking(masker, col="white")
