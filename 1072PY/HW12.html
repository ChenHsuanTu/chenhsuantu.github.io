import geopandas
TW=geopandas.read_file('data/TW.shp',encoding='utf-8')
POI=geopandas.read_file('data/POI.shp',encoding='utf-8')

TW['POI']=0
for i in range(len(POI)):
    for j in range(len(TW)):
        if POI.loc[i].geometry.within(TW.loc[j].geometry):
            TW.loc[j,'POI']=TW.loc[j,'POI']+1
            break
